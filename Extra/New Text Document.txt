letters = "abcefghikqdpaefh"


def generate_elements(str, todo):
    temp = []
    for c in todo:
        temp.append(str + c)
    return temp

def permutations_element(str, todo):
    if todo == "":
        print(str)
    else:
        permutations_list(generate_elements(str, todo), todo)

def permutations_list(los, todo):
    if len(los) == 0:
        return 
    else:
        element = los.pop(0)
        permutations_element(element, todo.replace(element[-1], ""))
        permutations_list(los, todo)


permutations_element("", letters)
